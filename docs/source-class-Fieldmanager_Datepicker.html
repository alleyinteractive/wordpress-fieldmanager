  7  * A Javascript date-picker which stores dates as unix timestamps.
  8  * @package Fieldmanager
  9  */
 10 class Fieldmanager_Datepicker extends Fieldmanager_Field {
 11 
 12     /**
 13      * @var boolean
 14      * Collect time info or just date info? Defaults to just date info.
 15      */
 16     public $use_time = False;
 17 
 18     /**
 19      * @var boolean
 20      * If true, and $use_time == true, and $date_element = 'dropdown', will render an 'AM' and 'PM' dropdown
 21      */
 22     public $use_am_pm = True;
 23 
 24     /**
 25      * @var string
 26      * PHP date format, only used for rendering already-saved dates. Use js_opts['dateFormat'] for the
 27      * date shown when a user selects an option. This option renders to '21 Apr 2013', and is fairly
 28      * friendly to international users.
 29      */
 30     public $date_format = 'j M Y';
 31 
 32     /**
 33      * @var array
 34      * Options to pass to the jQueryUI Datepicker. If you change dateFormat, be sure that it returns
 35      * a valid unix timestamp. Also, it's best to change js_opts['dateFormat'] and date_format together
 36      * for a consistent user experience.
 37      *
 38      * Default:
 39      * <code>
 40      * array(
 41      *   'showButtonPanel' => True,
 42      *   'showOtherMonths' => True,
 43      *   'selectOtherMonths' => True,
 44      *   'dateFormat' => 'd M yy',
 45      * );
 46      * </code>
 47      * @see http://api.jqueryui.com/datepicker/
 48      */
 49     public $js_opts = array();
 50 
 51     /**
 52      * Construct default attributes and enqueue javascript
 53      * @param array $options
 54      */
 55     public function __construct( $label, $options = array() ) {
 56         wp_enqueue_script( 'jquery-ui-datepicker' );
 57         fm_add_style( 'fm-jquery-ui', 'css/jquery-ui/jquery-ui-1.10.2.custom.min.css' );
 58         fm_add_script( 'fm_datepicker', 'js/fieldmanager-datepicker.js' );
 59         parent::__construct( $label, $options );
 60 
 61         if ( empty( $this->js_opts ) ) {
 62             $this->js_opts = array(
 63                 'showButtonPanel' => True,
 64                 'showOtherMonths' => True,
 65                 'selectOtherMonths' => True,
 66                 'dateFormat' => 'd M yy',
 67             );
 68         }
 69     }
 70 
 71     /**
 72      * Convert date to timestamp
 73      * @param $value
 74      * @param $current_value
 75      * @return int unix timestamp
 76      */
 77     public function presave( $value, $current_value = array() ) {
 78         $time_to_parse = sanitize_text_field( $value['date'] );
 79         if ( $this->use_time ) {
 80             $hour = intval( $value['hour'] );
 81             if ( $hour == 0 && $this->use_am_pm ) $hour = 12;
 82             $time_to_parse .= ' ' . $hour;
 83             $time_to_parse .= ':' . str_pad( intval( $value['minute'] ), 2, '0', STR_PAD_LEFT );
 84             $time_to_parse .= ' ' . sanitize_text_field( $value['ampm'] );
 85         }
 86         return intval( strtotime( $time_to_parse ) );
 87     }
 88 
 89     /**
 90      * Get hour for rendering in field
 91      * @param int $value unix timestamp
 92      * @return string value of hour
 93      */
 94     public function get_hour( $value ) {
 95         return !empty( $value ) ? date( $this->use_am_pm ? 'g' : 'G', $value ) : '';
 96     }
 97 
 98     /**
 99      * Get hour for rendering in field
100      * @param int $value unix timestamp
101      * @return string value of hour
102      */
103     public function get_minute( $value ) {
104         return !empty( $value ) ? date( 'i', $value ) : '';
105     }
106 
107     /**
108      * Get am or pm for rendering in field
109      * @param int $value unix timestamp
110      * @return string 'am', 'pm', or ''
111      */
112     public function get_am_pm( $value ) {
113         return ! empty( $value ) ? date( 'a', $value ) : '';
114     }
