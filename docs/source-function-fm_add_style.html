 50  * Add CSS and JS to admin area, hooked into admin_enqueue_scripts.

 52 function fieldmanager_enqueue_scripts() {
 53     wp_enqueue_script( 'fieldmanager_script', fieldmanager_get_baseurl() . 'js/fieldmanager.js', array( 'jquery' ), '1.0.1' );
 54     wp_enqueue_style( 'fieldmanager_style', fieldmanager_get_baseurl() . 'css/fieldmanager.css', array(), '1.0.0' );
 55     wp_enqueue_script( 'jquery-ui-sortable' );
 56 }

 68 /**
 69  * Get the base URL for this plugin.
 70  * @return string URL pointing to Fieldmanager top directory.
 71  */
 72 function fieldmanager_get_baseurl() {
 73     $path_override = _fieldmanager_registry( 'baseurl' );
 74     if ( $path_override ) {
 75         return $path_override;
 76     }
 77     return plugin_dir_url( __FILE__ );

 84 /**
 85  * Wrapper to enqueue_scripts which uses a closure
 86  * @param string $handle
 87  * @param string $path
 88  * @param string[] $deps
 89  * @param boolean $ver
 90  * @param boolean $in_footer
 91  * @param string $data_object
 92  * @param array $data
 93  * @param string $plugin_dir
 94  * @param boolean $admin
 95  * @return void
 96  */
 97 function fm_add_script( $handle, $path, $deps = array(), $ver = false, $in_footer = false, $data_object = "", $data = array(), $plugin_dir = "", $admin = true ) {
 98     if ( !is_admin() ) return;
 99     if ( !$ver ) $ver = FM_GLOBAL_ASSET_VERSION;
100     if ( $plugin_dir == "" ) $plugin_dir = fieldmanager_get_baseurl(); // allow overrides for child plugins
101     $add_script = function() use ( $handle, $path, $deps, $ver, $in_footer, $data_object, $data, $plugin_dir ) {
102         wp_enqueue_script( $handle, $plugin_dir . $path, $deps, $ver );
103         if ( !empty( $data_object ) && !empty( $data ) ) wp_localize_script( $handle, $data_object, $data );
104     };
105 
106     add_action( 'admin_enqueue_scripts', $add_script );
107     add_action( 'wp_enqueue_scripts', $add_script );
108 }

110 /**
111  * Wrapper to enqueue_style which uses a closure
112  * @param string $handle
113  * @param string $path
114  * @param string[] $deps
115  * @param boolean $ver
116  * @param string $media
117  * @param boolean $admin
118  * @return void
119  */
120 function fm_add_style( $handle, $path, $deps = array(), $ver = false, $media = 'all', $admin = true ) {
121     if( !is_admin() ) return;
122     if ( !$ver ) $ver = FM_GLOBAL_ASSET_VERSION;
123     $add_script = function() use ( $handle, $path, $deps, $ver, $media ) {
124         wp_register_style( $handle, fieldmanager_get_baseurl() . $path, $deps, $ver, $media );
125         wp_enqueue_style( $handle );
126     };
127 
128     add_action( 'admin_enqueue_scripts', $add_script );
129     add_action( 'wp_enqueue_scripts', $add_script );
130 }

132 /**
133  * A simple static registry for Fieldmanager to keep globals out.
134  * @param string $var what variable to set.
135  * @param mixed $val what value to store for $var.
136  * @return mixed if only $var is set, return stored $val. If both are set, set $val and return void.
137  */
138 function _fieldmanager_registry( $var, $val = NULL ) {
139     static $registry;
140     if ( !is_array( $registry ) ) $registry = array();
141     if ( $val === NULL ) {
142         return isset( $registry[$var] ) ? $registry[$var] : False;
143     }
144     $registry[$var] = $val;
145 }

158 /**
159  * Cheap way to tell if we're looking at a post edit page.
160  * It's a good idea to use this at the beginning of implementations which use metaboxes
161  * to avoid initializing your Fieldmanager chain on every page load.
162  * @return boolean are we editing or creating a post?
163  */
164 function fm_is_post_edit_screen() {
165     return stripos( $_SERVER['PHP_SELF'], '/post.php' ) !== FALSE || stripos( $_SERVER['PHP_SELF'], '/post-new.php' ) !== FALSE;
166 }

168 /**
169  * Sanitize multi-line text
170  * @param string $value unsanitized text
171  * @return string text with each line individually passed through sanitize_text_field.
172  */
173 function fm_sanitize_textarea( $value ) {
174     return implode( "\n", array_map( 'sanitize_text_field', explode( "\n", $value ) ) );
