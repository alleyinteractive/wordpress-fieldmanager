  6 /**
  7  * Base class containing core functionality for Fieldmanager
  8  *
  9  * Security features of this class for verifying access and saving data:
 10  * 1. In Fieldmanager_Field::save_fields_for_post(), verify that we are using an assigned post type
 11  *    This is the starting point for $_POST requests to post edit pages.
 12  * 2. In Fieldmanager_Field::save_fields_for_post(), verify that the user can save the post type
 13  * 3. In Fieldmanager_Field::save_fields_for_post(), perform nonce validation
 14  * 4. In Fieldmanager_Field::save_to_post_meta(), call Fieldmanager_Field::presave_all()
 15  *    This is the starting point for save routines that use Fieldmanager as an API
 16  * 5. In Fieldmanager_Field::presave_all(), verify that the number of elements is less than $limit,
 17  *    if $limit is not 1 or infinite (0), and verify that all multi-element arrays are numeric only.
 18  * 6. In Fieldmanager_Field::presave_all(), call Fieldmanager_Field::presave()
 19  * 5. In Fieldmanager_Field::presave(), call all assigned validators; if one returns false, die.
 20  * 6a. For groups (including the root group) in Fieldmanager_Group::presave(), verify that all
 21  *     keys in the submission match the form names of valid children.
 22  * 6b. For all other fields, in Fieldmanager_Field::presave(), sanitize the field to save to the DB.
 23  *     The default sanitizer is sanitize_text_field().
 24  *
 25  * @package Fieldmanager
 26  */

293     /**
294      * Build options into properties and throw errors if developers add an unsupported opt.
295      * @param string $label title of form field
296      * @param array $options with keys matching vars of the field in use.
297      * @throws FM_Developer_Exception if an option is set but not defined in this class or the child class.
298      * @throws FM_Developer_Exception if an option is set but not public.
299      */
300     public function set_options( $label, $options ) {
301         if ( is_array( $label ) ) $options = $label;
302         else $options['label'] = $label;
303 
304         foreach ( $options as $k => $v ) {
305             try {
306                 $reflection = new ReflectionProperty( $this, $k ); // Would throw a ReflectionException if item doesn't exist (developer error)
307                 if ( $reflection->isPublic() ) $this->$k = $v;
308                 else throw new FM_Developer_Exception; // If the property isn't public, don't set it (rare)
309             } catch ( Exception $e ) {
310                 $message = sprintf(
311                     __( 'You attempted to set a property <em>%1$s</em> that is nonexistant or invalid for an instance of <em>%2$s</em> named <em>%3$s</em>.' ),
312                     $k, __CLASS__, !empty( $options['name'] ) ? $options['name'] : 'NULL'
313                 );
314                 $title = __( 'Nonexistant or invalid option' );
315                 if ( !self::$debug ) {
316                     wp_die( $message, $title );
317                 } else {
318                     throw new FM_Developer_Exception( $message );
319                 }
320             }
321         }
322     }

324     /**
325      * Generates all markup needed for all form elements in this field.
326      * Could be called directly by a plugin or theme.
327      * @param array $values the current values of this element, in a tree structure if the element has children.
328      * @return string HTML for all form elements.
329      */
330     public function element_markup( $values = array() ) {
331         $values = $this->preload_alter_values( $values );
332         if ( $this->limit == 0 ) {
333             if ( count( $values ) + $this->extra_elements <= $this->starting_count ) {
334                 $max = $this->starting_count;
335             }
336             else {
337                 $max = count( $values ) + $this->extra_elements;
338             }
339         }
340         else {
341             $max = $this->limit;
342         }
343 
344         $classes = array( 'fm-wrapper', 'fm-' . $this->name . '-wrapper' );
345         $fm_wrapper_attrs = array();
346         if ( $this->sortable ) {
347             $classes[] = 'fmjs-sortable';
348         }
349         $classes = array_merge( $classes, $this->get_extra_element_classes() );
350 
351         $out = '';
352 
353         // If this element is part of tabbed output, there needs to be a wrapper to contain the tab content
354         if ( $this->is_tab ) {
355             $tab_display_style = ( $this->parent->child_count > 0 ) ? ' style="display: none"' : '';
356             $out .= '<div id="' . $this->get_element_id() . '-tab" class="wp-tabs-panel"' . $tab_display_style . '>';
357         }
358 
359         // For lists of items where $one_label_per_item = False, the label should go outside the wrapper.
360         if ( !empty( $this->label ) && !$this->one_label_per_item ) {
361             $out .= $this->get_element_label( array( 'fm-label-for-list' ) );
362         }
363 
364         // Find the array position of the "counter" (e.g. in element[0], [0] is the counter, thus the position is 1)
365         $html_array_position = 0; // default is no counter; i.e. if $this->limit = 0
366         if ( $this->limit != 1 ) {
367             $html_array_position = 1; // base situation is formname[0], so the counter is in position 1.
368             if ( $this->parent ) {
369                 $parent = $this->parent;
370                 while ( $parent ) {
371                     $html_array_position++; // one more for having a parent (e.g. parent[this][0])
372                     if ( $parent->limit != 1 ) { // and another for the parent having multiple (e.g. parent[0][this][0])
373                         $html_array_position++;
374                     }
375                     $parent = $parent->parent; // parent's parent; root element has null parent which breaks while loop.
376                 }
377             }
378         }
379 
380         // Checks to see if element has display_if data values, and inserts the data attributes if it does
381         if ( isset( $this->display_if ) && !empty( $this->display_if ) ) {
382             $classes[] = 'display-if';
383             $fm_wrapper_attrs['data-display-src'] = $this->display_if['src'];
384             $fm_wrapper_attrs['data-display-value'] = $this->display_if['value'];
385         }
386         $fm_wrapper_attr_string = '';
387         foreach ( $fm_wrapper_attrs as $attr => $val ) {
388             $fm_wrapper_attr_string .= sprintf( '%s="%s" ', $attr, htmlentities( $val ) );
389         }
390         $out .= sprintf( '<div class="%s" data-fm-array-position="%d" %s>',
391             implode( ' ', $classes ),
392             $html_array_position,
393             $fm_wrapper_attr_string
394         );
395 
396         // After starting the field, apply a filter to allow other plugins to append functionality
397         $out = apply_filters( 'fm_element_markup_start', $out, $this );
398 
399         for ( $i = 0; $i < $max; $i++ ) {
400             $this->seq = $i;
401             if ( $this->limit == 1 ) {
402                 $value = $values;
403             } else {
404                 $value = isset( $values[ $i ] ) ? $values[ $i ] : Null;
405             }
406             $out .= $this->single_element_markup( $value );
407         }
408         if ( $this->limit == 0 ) {
409             $out .= $this->add_another();
410         }
411 
412         // Before closing the field, apply a filter to allow other plugins to append functionality
413         $out = apply_filters( 'fm_element_markup_end', $out, $this );
414 
415         $out .= '</div>';
416 
417         // Close the tab wrapper if one exists
418         if ( $this->is_tab ) $out .= '</div>';
419 
420         return $out;
421     }

423     /**
424      * Generate wrappers and labels for one form element.
425      * Is called by element_markup(), calls form_element().
426      * @see Fieldmanager_Field::element_markup()
427      * @see Fieldmanager_Field::form_element()
428      * @param mixed $value the current value of this element.
429      * @param boolean $is_proto true to generate a prototype element for Javascript.
430      * @return string HTML for a single form element.
431      */
432     public function single_element_markup( $value = Null, $is_proto = False ) {
433         $out = '';
434         $classes = array( 'fm-item', 'fm-' . $this->name );
435 
436         self::$global_seq++;
437 
438         // Drop the fm-group class to hide inner box display if no label is set
439         if ( !( $this->field_class == 'group' && ( !isset( $this->label ) || empty( $this->label ) ) ) ) {
440             $classes[] = 'fm-' . $this->field_class;
441         }
442 
443         // Check if the required attribute is set. If so, add the class.
444         if ( $this->required ) {
445             $classes[] = 'form-required';
446         }
447 
448         if ( $this->get_seq() == 0 && $this->limit == 0 ) {
449             // Generate a prototype element for DOM magic on the frontend.
450             if ( $is_proto ) {
451                 $classes[] = 'fmjs-proto';
452             } else {
453                 $this->is_proto = True;
454                 $out .= $this->single_element_markup( Null, True );
455                 $this->is_proto = False;
456             }
457         }
458 
459         $out .= sprintf( '<div class="%s">', implode( ' ', $classes ) );
460 
461         $label = $this->get_element_label( );
462         $render_label_after = False;
463         // Hide the label if it is empty or if this is a tab since it would duplicate the title from the tab label
464         if ( !empty( $this->label ) && !$this->is_tab && $this->one_label_per_item ) {
465             if ( $this->limit == 0 && $this->one_label_per_item ) {
466                 $out .= $this->wrap_with_multi_tools( $label, array( 'fmjs-removable-label' ) );
467             } elseif ( !$this->label_after_element ) {
468                 $out .= $label;
469             } else {
470                 $render_label_after = True;
471             }
472         }
473 
474         if ( Null === $value && Null !== $this->default_value )
475             $value = $this->default_value;
476 
477         $form_element = $this->form_element( $value );
478 
479         if ( $this->limit == 0 && !$this->one_label_per_item ) {
480             $out .= $this->wrap_with_multi_tools( $form_element );
481         } else {
482             $out .= $form_element;
483         }
484 
485         if ( $render_label_after ) $out .= $label;
486 
487         if ( isset( $this->description ) && !empty( $this->description ) ) {
488             $out .= sprintf( '<div class="fm-item-description">%s</div>', $this->description );
489         }
490 
491         $out .= '</div>';
492         return $out;
493     }

555     /**
556      * Get the ID for the form element itself, uses $this->seq (e.g. which position is this element in).

571     /**
572      * Presaves all elements in what could be a set of them, dispatches to $this->presave()
573      * @input mixed[] $values
574      * @return mixed[] sanitized values
575      */
576     public function presave_all( $values, $current_values ) {
577 
578         if ( $this->limit == 1 && empty( $this->multiple ) ) {
579             $values = $this->presave_alter_values( array( $values ), array( $current_values ) );
580             if ( ! empty( $values ) )
581                 $value = $this->presave( $values[0], $current_values );
582             else
583                 $value = $values;
584             if ( !empty( $this->index ) ) $this->save_index( array( $value ), array( $current_values ) );
585             return $value;
586         }
587 
588         // If $this->limit != 1, and $values is not an array, that'd just be wrong, and possibly an attack, so...
589         if ( $this->limit != 1 && !is_array( $values ) ) {
590             $this->_unauthorized_access( '$values should be an array because $limit is ' . $this->limit );
591         }
592 
593         // If $this->limit is not 0 or 1, and $values has more than $limit, that could also be an attack...
594         if ( $this->limit > 1 && count( $values ) > $this->limit ) {
595             $this->_unauthorized_access(
596                 sprintf( 'submitted %1$d values against a limit of %2$d', count( $values ), $this->limit )
597             );
598         }
599 
600         if ( empty( $values ) ) $values = array();
601 
602         if ( isset( $values['proto'] ) ) {
603             unset( $values['proto'] );
604         }
605 
606         # Condense the array to account for middle items removed
607         $values = array_values( $values );
608 
609         $values = $this->presave_alter_values( $values, $current_values );
610 
611         foreach ( $values as $i => $value ) {
612             $values[ $i ] = $this->presave( $value, empty( $current_values[ $i ] ) ? array() : $current_values[ $i ] );
613             if ( !$this->save_empty && empty( $values[ $i ] ) )
614                 unset( $values[ $i ] );
615         }
616         if ( !empty( $this->index ) )
617             $this->save_index( $values, $current_values );
618 
619         return $values;
620     }

622     /**
623      * Optionally save fields to a separate postmeta index for easy lookup with WP_Query
624      * Handles internal arrays (e.g. for fieldmanager-options).
625      * Is called multiple times for multi-fields (e.g. limit => 0)
626      * @param array $values
627      * @return void
628      * @todo make this a context method
629      */
630     protected function save_index( $values, $current_values ) {
631         if ( $this->data_type != 'post' || empty( $this->data_id ) ) return;
632         // Must delete current values specifically, then add new ones, to support a scenario where the
633         // same field in repeating groups with limit = 1 is going to create more than one entry here, and
634         // if we called update_post_meta() we would overwrite the index with each new group.
635         foreach ( $current_values as $old_value ) {
636             if ( !is_array( $old_value ) ) $old_value = array( $old_value );

666     public function presave( $value, $current_value = array() ) {
667         // It's possible that some elements (Grid is one) would be arrays at
668         // this point, but those elements must override this function. Let's
669         // make sure we're dealing with one value here.
670         if ( is_array( $value ) ) {
671             $this->_unauthorized_access( 'presave() in the base class should not get arrays, but did.' );
