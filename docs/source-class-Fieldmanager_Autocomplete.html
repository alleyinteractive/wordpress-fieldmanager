 91     /**
 92      * Render form element
 93      * @param mixed $value
 94      * @return string HTML

 96     public function form_element( $value = Null ) {
 97 
 98         if ( $this->exact_match ) {
 99             $this->attributes['data-exact-match'] = True;
100         }
101 
102         if ( $this->datasource->use_ajax ) {
103             $this->attributes['data-action'] = $this->datasource->get_ajax_action( $this->name );
104         } else {
105             $this->attributes['data-options'] = htmlspecialchars( json_encode( $this->datasource->options ) );
106         }
107 
108         $element = sprintf(
109             '<input class="fm-autocomplete fm-element" type="text" id="%s" value="%s" %s />',
110             $this->get_element_id(),
111             $this->datasource->get_value( $value ),
112             $this->get_element_attributes()
113         );
114 
115         $element .= sprintf(
116             '<input class="fm-autocomplete-hidden fm-element" type="hidden" name="%s" value="%s" />',
117             $this->get_form_name(),
118             $value
119         );
120 
121         if ( isset( $this->show_view_link ) && $this->show_view_link ) {
122             $element .= $this->datasource->get_view_link( $value );
123         }
124 
125         if ( isset( $this->show_edit_link ) && $this->show_edit_link ) {
126             $element .= $this->datasource->get_edit_link( $value );
127         }
128 
129         return $element;
130     }

152     /**
153      * Helper function to get the list of default meta boxes to remove.
154      * If $remove_default_meta_boxes is true and the datasource is Fieldmanager_Datasource_Term,
155      * this will return a list of all default meta boxes for the specified taxonomies.
156      * We only need to return id and context since the page will be handled by the list of post types provided to add_meta_box.
157      * Otherwise, this will just return an empty array.
158      * @param array current list of meta boxes to remove
159      * @return array list of meta boxes to remove
160      */
161     protected function add_meta_boxes_to_remove( &$meta_boxes_to_remove ) {
162         if ( $this->remove_default_meta_boxes && get_class( $this->datasource ) == 'Fieldmanager_Datasource_Term' ) {
163             // Iterate over the list and build the list of meta boxes
164             $meta_boxes = array();
165             foreach( $this->datasource->get_taxonomies() as $taxonomy ) {
166                 // The ID differs if this is a hierarchical taxonomy or not. Get the taxonomy object.
167                 $taxonomy_obj = get_taxonomy( $taxonomy );
168                 if ( false !== $taxonomy_obj ) {
169                     if ( $taxonomy_obj->hierarchical )
170                         $id = $taxonomy . "div";
171                     else
172                         $id = 'tagsdiv-' . $taxonomy;
173 
174                     $meta_boxes[$id] = array(
175                         'id' => $id,
176                         'context' => 'side'
177                     );
178                 }
179             }
180 
181             // Merge in the new meta boxes to remove
182             $meta_boxes_to_remove = array_merge( $meta_boxes_to_remove, $meta_boxes );
183         }
184     }
