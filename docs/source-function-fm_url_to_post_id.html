 65     /**
 66      * Get posts which match this datasource, optionally filtered by 
 67      * a fragment, e.g. for Autocomplete.
 68      * @param string $fragment
 69      * @return array post_id => post_title for display or AJAX
 70      */
 71     public function get_items( $fragment = Null ) {
 72         if ( is_callable( $this->query_callback ) ) {
 73             return call_user_func( $this->query_callback, $fragment );
 74         }
 75         $default_args = array(
 76             'numberposts' => 10,
 77             'orderby' => 'post_date',
 78             'order' => 'DESC',
 79             'post_status' => 'publish',
 80             'post_type' => 'any',
 81             'suppress_filters' => False,
 82         );
 83         $post_args = array_merge( $default_args, $this->query_args );
 84         $ret = array();
 85         if ( $fragment ) {
 86             $post_id = $exact_post = Null;
 87             if ( preg_match( '/^https?\:/i', $fragment ) ) {
 88                 $url = esc_url( $fragment );
 89                 $url_parts = parse_url( $url );
 90                 $get_vars = array();
 91                 parse_str( $url_parts['query'], $get_vars );
 92                 if ( !empty( $get_vars['post'] )  ) {
 93                     $post_id = intval( $get_vars['post'] );
 94                 } elseif ( !empty( $get_vars['p'] ) ) {
 95                     $post_id = intval( $get_vars['p'] );
 96                 } else {
 97                     $post_id = fm_url_to_post_id( $fragment );
 98                 }
 99             } elseif ( is_numeric( $fragment ) ) {
100                 $post_id = intval( $fragment );
101             }
102             if ( $post_id ) {
103                 $exact_post = get_post( $post_id );
104                 if ( $exact_post && (
105                     $post_args['post_type'] == 'any' ||
106                     $post_args['post_type'] == $exact_post->post_type ||
107                     in_array( $exact_post->post_type, $post_args['post_type'] )
108                 ) ) {
109                     if ( $this->show_date ) {
110                         $date_pad = ' (' . date( $this->date_format, strtotime( $exact_post->post_date ) ) . ')';
111                     }
112                     else {
113                         $date_pad = '';
114                     }
115                     $ret[ $post_id ] = html_entity_decode( $exact_post->post_title ) . $date_pad;
116                 }
117             }
118             $this->_fragment = $fragment;
119             add_filter( 'posts_where', array( $this, 'title_like' ), 10, 2 );
120         }
121         $posts = get_posts( $post_args );
122         if ( $fragment ) {
123             remove_filter( 'posts_where', array( $this, 'title_like' ), 10, 2 );
124         }
125         foreach ( $posts as $p ) {
126             if ( $this->show_date ) {
127                 $date_pad = ' (' . date( $this->date_format, strtotime( $p->post_date ) ) . ')';
128             }
129             else {
130                 $date_pad = '';
131             }
132             $ret[$p->ID] = $p->post_title . $date_pad;
133         }
134         return $ret;
135     }

158     /**
159      * For post relationships, delete reciprocal post metadata prior to saving (presave will re-add)
160      * @param array $values new post values
161      * @param array $current_values existing post values
162      */
163     public function presave_alter_values( Fieldmanager_Field $field, $values, $current_values ) {
164         if ( $field->data_type != 'post' || !$this->reciprocal ) return $values;
165         foreach ( $current_values as $reciprocal_post_id ) {
166             delete_post_meta( $reciprocal_post_id, $this->reciprocal, $field->data_id );
167         }
168         return $values;
169     }

171     /**
172      * Handle reciprocal postmeta
173      * @param int $value
174      * @return string
175      */
176     public function presave( Fieldmanager_Field $field, $value, $current_value ) {
177         if ( empty( $value ) ) return;
178         $value = intval( $value );
179         if( !current_user_can( 'edit_post', $value ) ) {
180             die( 'Tried to refer to post ' . $value . ' which user cannot edit.' );  
181         }
182         if ( $this->reciprocal ) {
183             add_post_meta( $value, $this->reciprocal, $field->data_id );
184         }
185         return $value;
186     }

218 /**
219  * Post URLs to IDs function, supports custom post types.
220  * Borrowed and modified from url_to_postid() in wp-includes/rewrite.php
221  * @author http://betterwp.net/
222  * @param string $url
223  * @return int|boolean post ID on success, false on failure
224  */
225 function fm_url_to_post_id( $url ) {
226     global $wp_rewrite;
227  
228     $url = apply_filters('url_to_postid', $url);
229  
230     // First, check to see if there is a 'p=N' or 'page_id=N' to match against
231     if ( preg_match('#[?&](p|page_id|attachment_id)=(\d+)#', $url, $values ) ) {
232         $id = absint($values[2]);
233         if ( $id )
234             return $id;
235     }
236  
237     // Check to see if we are using rewrite rules
238     $rewrite = $wp_rewrite->wp_rewrite_rules();
239  
240     // Not using rewrite rules, and 'p=N' and 'page_id=N' methods failed, so we're out of options
241     if ( empty( $rewrite ) )
242         return 0;
243  
244     // Get rid of the #anchor
245     $url_split = explode( '#', $url );
246     $url = $url_split[0];
247  
248     // Get rid of URL ?query=string
249     $url_split = explode('?', $url);
250     $url = $url_split[0];
251  
252     // Add 'www.' if it is absent and should be there
253     if ( false !== strpos(home_url(), '://www.') && false === strpos($url, '://www.') )
254         $url = str_replace('://', '://www.', $url);
255  
256     // Strip 'www.' if it is present and shouldn't be
257     if ( false === strpos(home_url(), '://www.') )
258         $url = str_replace('://www.', '://', $url);
259  
260     // Strip 'index.php/' if we're not using path info permalinks
261     if ( !$wp_rewrite->using_index_permalinks() )
262         $url = str_replace('index.php/', '', $url);
263  
264     if ( false !== strpos($url, home_url()) ) {
265         // Chop off http://domain.com
266         $url = str_replace(home_url(), '', $url);
267     } else {
268         // Chop off /path/to/blog
269         $home_path = parse_url(home_url());
270         $home_path = isset( $home_path['path'] ) ? $home_path['path'] : '' ;
271         $url = str_replace($home_path, '', $url);
272     }
273  
274     // Trim leading and lagging slashes
275     $url = trim($url, '/');
276  
277     $request = $url;
278     // Look for matches.
279     $request_match = $request;
280     foreach ( (array)$rewrite as $match => $query) {
281         // If the requesting file is the anchor of the match, prepend it
282         // to the path info.
283         if ( !empty($url) && ($url != $request) && (strpos($match, $url) === 0) )
284             $request_match = $url . '/' . $request;
285  
286         if ( preg_match("!^$match!", $request_match, $matches) ) {
287             // Got a match.
288             // Trim the query of everything up to the '?'.
289             $query = preg_replace("!^.+\?!", '', $query);
290  
291             // Substitute the substring matches into the query.
292             $query = addslashes(WP_MatchesMapRegex::apply($query, $matches));
293  
294             // Filter out non-public query vars
295             global $wp;
296             parse_str($query, $query_vars);
297             $query = array();
298             foreach ( (array) $query_vars as $key => $value ) {
299                 if ( in_array($key, $wp->public_query_vars) )
300                     $query[$key] = $value;
301             }
302  
303         // Taken from class-wp.php
304         foreach ( $GLOBALS['wp_post_types'] as $post_type => $t )
305             if ( $t->query_var )
306                 $post_type_query_vars[$t->query_var] = $post_type;
307  
308         foreach ( $wp->public_query_vars as $wpvar ) {
309             if ( isset( $wp->extra_query_vars[$wpvar] ) )
310                 $query[$wpvar] = $wp->extra_query_vars[$wpvar];
311             elseif ( isset( $_POST[$wpvar] ) )
312                 $query[$wpvar] = $_POST[$wpvar];
313             elseif ( isset( $_GET[$wpvar] ) )
314                 $query[$wpvar] = $_GET[$wpvar];
315             elseif ( isset( $query_vars[$wpvar] ) )
316                 $query[$wpvar] = $query_vars[$wpvar];
317  
318             if ( !empty( $query[$wpvar] ) ) {
319                 if ( ! is_array( $query[$wpvar] ) ) {
320                     $query[$wpvar] = (string) $query[$wpvar];
321                 } else {
322                     foreach ( $query[$wpvar] as $vkey => $v ) {
323                         if ( !is_object( $v ) ) {
324                             $query[$wpvar][$vkey] = (string) $v;
325                         }
326                     }
327                 }
328  
329                 if ( isset($post_type_query_vars[$wpvar] ) ) {
330                     $query['post_type'] = $post_type_query_vars[$wpvar];
331                     $query['name'] = $query[$wpvar];
332                 }
333             }
334         }
335  
336             // Do the query
337             $query = new WP_Query($query);
338             if ( !empty($query->posts) && $query->is_singular )
339                 return $query->post->ID;
340             else
341                 return 0;
342         }
343     }
344     return 0;
