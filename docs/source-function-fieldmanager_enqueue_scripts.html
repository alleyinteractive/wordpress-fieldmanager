 65: /**
 66:  * Wrapper to enqueue_scripts which uses a closure
 67:  * @param string $handle
 68:  * @param string $path
 69:  * @param string[] $deps
 70:  * @param boolean $ver
 71:  * @param boolean $in_footer
 72:  * @param string $data_object
 73:  * @param array $data
 74:  * @param string $plugin_dir
 75:  * @return void
 76:  */
 77: function fm_add_script( $handle, $path, $deps = array(), $ver = false, $in_footer = false, $data_object = "", $data = array(), $plugin_dir = "" ) {
 78:     if ( !is_admin() ) return;
 79:     if ( !$ver ) $ver = FM_GLOBAL_ASSET_VERSION;
 80:     if ( $plugin_dir == "" ) $plugin_dir = fieldmanager_get_baseurl(); // allow overrides for child plugins
 81:     $add_script = function() use ( $handle, $path, $deps, $ver, $in_footer, $data_object, $data, $plugin_dir ) {
 82:         wp_enqueue_script( $handle, $plugin_dir . $path, $deps, $ver );
 83:         if ( !empty( $data_object ) && !empty( $data ) ) wp_localize_script( $handle, $data_object, $data );
 84:     };
 85:     add_action( 'admin_enqueue_scripts', $add_script );
 86: }

107: /**
108:  * A simple static registry for Fieldmanager to keep globals out.
109:  * @param string $var what variable to set.
110:  * @param mixed $val what value to store for $var.
111:  * @return mixed if only $var is set, return stored $val. If both are set, set $val and return void.
112:  */
113: function _fieldmanager_registry( $var, $val = NULL ) {
114:     static $registry;
115:     if ( !is_array( $registry ) ) $registry = array();
116:     if ( $val === NULL ) {
117:         return isset( $registry[$var] ) ? $registry[$var] : False;
118:     }
119:     $registry[$var] = $val;
120: }
