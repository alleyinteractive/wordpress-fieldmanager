 20 /**
 21  * Loads a class based on classname. Understands Fieldmanager nomenclature to find the right file.
 22  * Does not know how to load classes outside of Fieldmanager's plugin, so if you build your own field,
 23  * you will have to include it or autoload it for yourself.
 24  * @uses spl_autoload_register
 25  * @uses fieldmanager_load_file
 26  * @param string $class
 27  */
 28 function fieldmanager_load_class( $class ) {
 29     static $first_load = true;
 30 
 31     if ( class_exists( $class ) || strpos( $class, 'Fieldmanager' ) !== 0 ) return;
 32     $class_id = strtolower( substr( $class, strrpos( $class, '_' ) + 1 ) );
 33 
 34     if ( $first_load ) {
 35         // Utility classes with helper functions
 36         fieldmanager_load_file( 'util/class-fieldmanager-util-term-meta.php' );
 37         fieldmanager_load_file( 'util/class-fieldmanager-util-validation.php' );
 38         $first_load = false;
 39     }
 40 
 41     if ( strpos( $class, 'Fieldmanager_Context' ) === 0 ) {
 42         if ( $class_id == 'context' ) return fieldmanager_load_file( 'context/class-fieldmanager-context.php' );    
 43         return fieldmanager_load_file( 'context/class-fieldmanager-context-' . $class_id . '.php' );
 44     }
 45 
 46     if ( strpos( $class, 'Fieldmanager_Datasource' ) === 0 ) {
 47         if ( $class_id == 'datasource' ) return fieldmanager_load_file( 'datasource/class-fieldmanager-datasource.php' );
 48         return fieldmanager_load_file( 'datasource/class-fieldmanager-datasource-' . $class_id . '.php' );
 49     }
 50     return fieldmanager_load_file( 'class-fieldmanager-' . $class_id . '.php', $class );
 51 }

153 /**
154  * A simple static registry for Fieldmanager to keep globals out.
155  * @param string $var what variable to set.
156  * @param mixed $val what value to store for $var.
157  * @return mixed if only $var is set, return stored $val. If both are set, set $val and return void.
158  */
159 function _fieldmanager_registry( $var, $val = NULL ) {
160     static $registry;
161     if ( !is_array( $registry ) ) $registry = array();
162     if ( $val === NULL ) {
163         return isset( $registry[$var] ) ? $registry[$var] : False;
164     }
165     $registry[$var] = $val;
166 }

168 /**
169  * Get the context for triggers and pattern matching.
170  *
171  * This function is crucial for performance. It prevents the unnecessary initialization of FM classes,
172  * and the unnecessary loading of CSS and Javascript.
173  * 
174  * You can't use this function to determine whether or not a context 'Form' will be displayed, since
175  * it can be used anywhere. We would love to use get_current_screen(), but it's not available in
176  * some POST actions, and generally not available early enough in the init process.
177  *
178  * This is a function to watch closely as WordPress changes, since it relies on paths and variables.
179  *
180  * @return string[] [$context, $type]
181  */
182 function fm_get_context() {
183     static $calculated_context;
184 
185     if ( $calculated_context ) return $calculated_context;
186 
187     if ( is_admin() ) { // safe to use at any point in the load process, and better than URL matching.
188 
189         $script = substr( $_SERVER['PHP_SELF'], strrpos( $_SERVER['PHP_SELF'], '/' ) + 1 );
190 
191         // context = submenu
192         if ( !empty( $_GET['page'] ) ) {
193             $submenus = _fieldmanager_registry( 'submenus' );
194             if ( $submenus ) {
195                 foreach ( $submenus as $submenu ) {
196                     if ( $script == $submenu[0] ) {
197                         $calculated_context = array( 'submenu', sanitize_text_field( $_GET['page'] ) );
198                         return $calculated_context;
199                     }
200                 }
201             }
202         }
203 
204         switch ( $script ) {
205             // context = post
206             case 'post.php':
207                 if ( !empty( $_POST['action'] ) && ( 'editpost' === $_POST['action'] || 'newpost' === $_POST['action'] ) ) {
208                     $calculated_context = array( 'post', sanitize_text_field( $_POST['post_type'] ) );
209                 } elseif ( !empty( $_GET['post'] ) ) {
210                     $calculated_context = array( 'post', get_post_type( intval( $_GET['post'] ) ) );
211                 }
212                 break;
213             case 'post-new.php':
214                 $calculated_context = array( 'post', !empty( $_GET['post_type'] ) ? sanitize_text_field( $_GET['post_type'] ) : 'post' );
215                 break;
216             // context = user
217             case 'profile.php':
218             case 'user-edit.php':
219                 $calculated_context = array( 'user', null );
220                 break;
221             // context = quickedit
222             case 'edit.php':
223                 $calculated_context = array( 'quickedit', !empty( $_GET['post_type'] ) ? sanitize_text_field( $_GET['post_type'] ) : 'post' );
224                 break;
225             case 'admin-ajax.php':
226                 // passed in via an ajax form
227                 if ( !empty( $_POST['fm_context'] ) ) {
228                     $subcontext = !empty( $_POST['fm_subcontext'] ) ? sanitize_text_field( $_POST['fm_subcontext'] ) : null;
229                     $calculated_context = array( sanitize_text_field( $_POST['fm_context'] ), $subcontext );
230                 } elseif ( !empty( $_POST['screen'] ) && !empty( $_POST['action'] ) ) {
231                     if ( 'edit-post' === $_POST['screen'] && 'inline-save' === $_POST['action'] ) {
232                         $calculated_context = array( 'quickedit', sanitize_text_field( $_POST['post_type'] ) );
233                     // context = term
234                     } elseif ( 'add-tag' === $_POST['action'] && !empty( $_POST['taxonomy'] ) ) {
235                         $calculated_context = array( 'term', sanitize_text_field( $_POST['taxonomy'] ) );
236                     }
237                 // context = quickedit
238                 } elseif ( !empty( $_GET['action'] ) && 'fm_quickedit_render' === $_GET['action'] ) {
239                     $calculated_context = array( 'quickedit', sanitize_text_field( $_GET['post_type'] ) );  
240                 }
241                 break;
242             // context = term
243             case 'edit-tags.php':
244                 if ( !empty( $_POST['taxonomy'] ) ) {
245                     $calculated_context = array( 'term', sanitize_text_field( $_POST['taxonomy'] ) );
246                 } elseif ( !empty( $_GET['taxonomy'] ) ) {
247                     $calculated_context = array( 'term', sanitize_text_field( $_GET['taxonomy'] ) );
248                 }
249                 break;
250         }
251     }
252 
253     if ( empty( $calculated_context ) ) {
254         $calculated_context = array( null, null );
255     }
256     return $calculated_context;
257 }

259 /**
260  * Check to see if a given context is active.
261  * @see fm_get_context()
262  * @param string $context one of 'post', 'quickedit', 'submenu', 'term', 'form', or 'user'.
263  * @param string|string[] $type Optional. For 'post' and 'quickedit', pass the post type. For 'term' it will be
264  *   the taxonomy. For 'submenu' it will be the page name. For all others it will be null. You can pass an array
265  *   of types to match several types.
266  * @return string[] - a two-element array, like [context, type].
267  */
268 function fm_match_context( $context, $type = null ) {
269     if ( $context == 'form' ) return true; // nothing to check, since forms can be anywhere.
270     $calculated_context = fm_get_context();
271     if ( $context == $calculated_context[0] ) {
272         if ( $type !== null ) {
273             if ( is_array( $type ) ) {
274                 return in_array( $calculated_context[1], $type );
275             }
276             return ( $type == $calculated_context[1] );
277         }
278         return true;
279     }
280     return false;
281 }

296 /**
297  * Register a submenu page
298  * @throws FM_Duplicate_Submenu_Name
299  */
300 function fm_register_submenu_page( $group_name, $parent_slug, $page_title, $menu_title = Null, $capability = 'manage_options', $menu_slug = Null ) {
301     $submenus = _fieldmanager_registry( 'submenus' );
302     if ( !$submenus ) $submenus = array();
303     if ( isset( $submenus[ $group_name ] ) ) {
304         throw new FM_Duplicate_Submenu_Name_Exception( $group_name . ' is already in use as a submenu name' );
305     }
306 
307     if ( !$menu_title ) $menu_title = $page_title;
308 
309     // will be replaced by a Fieldmanager_Context_Submenu if this submenu page is active.
310     $submenus[ $group_name ] = array( $parent_slug, $page_title, $menu_title, $capability, $menu_slug ?: $group_name, '_fm_submenu_render' );
311 
312     _fieldmanager_registry( 'submenus', $submenus );
313 }

318 function _fm_submenu_render() {
319     $context = _fieldmanager_registry( 'active_submenu' );
320     if ( !is_object( $context ) ) {
