 50      * Add CSS, construct parent

 54     public function __construct( $label = '', $options = array() ) {
 55         parent::__construct( $label, $options );
 56 
 57         if ( !empty( $this->options ) ) {
 58             $this->add_options( $this->options );

 63         
 64         // Sanitization
 65         $this->sanitize = function( $value ) {
 66             if ( isset( $value ) && is_array( $value ) && !empty( $value ) ) {
 67                 return array_map( 'sanitize_text_field', $value );
 68             } else {
 69                 return sanitize_text_field( $value );

 74     /**
 75      * Add options
 76      * @param array $options
 77      * @return void
 78      */
 79     public function add_options( $options ) {
 80         $values = array_values( $options );
 81         if ( is_array( $values[0] ) ) {
 82             foreach ( $options as $group => $data ) {
 83                 foreach ( $data as $value => $label ) {
 84                     $this->add_option_data( $value, $label, $group, $group );
 85                 }
 86             }
 87         } else {
 88             $keys = array_keys( $options );
 89             $use_name_as_value = ( array_keys( $keys ) === $keys );
 90             foreach ( $options as $k => $v ) {
 91                 $this->add_option_data( $v, ( $use_name_as_value ? $v : $k ) );
 92             }
 93         }
 94     }

 96     /**
 97      * Generate form elements.
 98      * @param mixed $value
 99      * @return string HTML
100      */
101     public function form_data_elements( $value ) {
102     
103         if ( !$this->has_built_data ) {
104             if ( $this->datasource ) {
105                 $this->add_options( $this->datasource->get_items() );
106             }
107         
108             // Add the first element to the data array. This is useful for database-based data sets that require a first element.
109             if ( !empty( $this->first_element ) ) array_unshift( $this->data, $this->first_element );
110             $this->has_built_data = True;
111         }
112         
113         // If the value is not in an array, put it in one since sometimes there will be multiple selects
114         if ( !is_array( $value ) && isset( $value ) ) {
115             $value = array( $value );
116         }
117     
118         // Output the data for the form. Child classes will handle the output format appropriate for them.
119         $form_data_elements_html = '';
120 
121         if ( !empty( $this->data ) ) {
122         
123             $current_group = '';
124 
125             foreach( $this->data as $data_element ) {
126             
127                 // If grouped display is desired, check where to add the start and end points
128                 // Note we are assuming the data has come pre-sorted into groups
129                 if( $this->grouped && ( $current_group != $data_element['group'] ) ) {
130                     
131                     // Append the end for the previous group unless this is the first group 
132                     if ( $current_group != '' ) $form_data_elements_html .= $this->form_data_end_group();
133                     
134                     // Append the start of the group
135                     $form_data_elements_html .= $this->form_data_start_group( $data_element['group'] );
136                     
137                     // Set the current group
138                     $current_group = $data_element['group'];
139                 }
140                 
141                 // Get the current element
142                 $form_data_elements_html .= $this->form_data_element( $data_element, $value );
143             }
144             
145             // If this was grouped display, close the final group
146             if( $this->grouped || ( isset( $this->datasource ) && $this->datasource->grouped ) ) $form_data_elements_html .= $this->form_data_end_group();
147         }
148 
149         return $form_data_elements_html;
150     
151     }

179     /**
180      * Presave function, which handles sanitization and validation
181      * @param mixed $value If a single field expects to manage an array, it must override presave()
182      * @return sanitized values. 
183      */
184     public function presave( $value, $current_value = array() ) {
185         if ( !empty( $this->datasource ) ) {
186             return $this->datasource->presave( $this, $value, $current_value );
187         }
188         foreach ( $this->validate as $func ) {
189             if ( !call_user_func( $func, $value ) ) {
190                 $this->_failed_validation( sprintf(
191                     __( 'Input "%1$s" is not valid for field "%2$s" ' ),
192                     (string) $value,
193                     $this->label
194                 ) );
195             }
196         }
197         return call_user_func( $this->sanitize, $value );
198     }

232         $data = array(

242     /**
243      * Helper function to get the list of default meta boxes to remove.
244      * If $remove_default_meta_boxes is true and the datasource is Fieldmanager_Datasource_Term, 
245      * this will return a list of all default meta boxes for the specified taxonomies.
246      * We only need to return id and context since the page will be handled by the list of post types provided to add_meta_box.
247      * Otherwise, this will just return an empty array.
248      * @param array current list of meta boxes to remove
249      * @return array list of meta boxes to remove
250      */
251     protected function add_meta_boxes_to_remove( &$meta_boxes_to_remove ) {
252         if ( $this->remove_default_meta_boxes && get_class( $this->datasource ) == 'Fieldmanager_Datasource_Term' ) {
253             // Iterate over the list and build the list of meta boxes
254             $meta_boxes = array();
255             foreach( $this->datasource->get_taxonomies() as $taxonomy ) {
256                 // The ID differs if this is a hierarchical taxonomy or not. Get the taxonomy object.
257                 $taxonomy_obj = get_taxonomy( $taxonomy );
258                 if ( false !== $taxonomy_obj ) {
259                     if ( $taxonomy_obj->hierarchical )
260                         $id = $taxonomy . "div";
261                     else
262                         $id = 'tagsdiv-' . $taxonomy;
263                     
264                     $meta_boxes[$id] = array(
265                         'id' => $id,
266                         'context' => 'side'
267                     );
268                 }
269             }
270             
271             // Merge in the new meta boxes to remove
272             $meta_boxes_to_remove = array_merge( $meta_boxes_to_remove, $meta_boxes );
273         }
274     }
