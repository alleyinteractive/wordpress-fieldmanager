 47     public function __construct( $options = array() ) {
 48 
 49         foreach ( $options as $k => $v ) {
 50             try {
 51                 $reflection = new ReflectionProperty( $this, $k ); // Would throw a ReflectionException if item doesn't exist (developer error)
 52                 if ( $reflection->isPublic() ) $this->$k = $v;
 53                 else throw new FM_Developer_Exception; // If the property isn't public, don't set it (rare)
 54             } catch ( Exception $e ) {
 55                 $message = sprintf(
 56                     __( 'You attempted to set a property <em>%1$s</em> that is nonexistant or invalid for an instance of <em>%2$s</em> named <em>%3$s</em>.' ),
 57                     $k, __CLASS__, !empty( $options['name'] ) ? $options['name'] : 'NULL'
 58                 );
 59                 $title = __( 'Nonexistant or invalid option' );
 60                 if ( !Fieldmanager_Field::$debug ) {
 61                     wp_die( $message, $title );
 62                 } else {
 63                     throw new FM_Developer_Exception( $message );
 64                 }
 65             }
 66         }
 67 
 68         if ( get_class( $this ) == __CLASS__ && empty( $options ) ) {
 69             $message = __( 'Invalid options for Datasource; must use the options parameter to supply an array.' );
 70             if ( Fieldmanager_Field::$debug ) {
 71                 throw new FM_Developer_Exception( $message );
 72             } else {
 73                 wp_die( $message, __( 'Invalid Datasource Options' ) );
 74             }
 75         }
 76 
 77         if ( !empty( $this->options ) ) {
 78             $keys = array_keys( $this->options );
 79             $use_name_as_value = ( array_keys( $keys ) === $keys );
 80             foreach ( $this->options as $k => $v ) {
 81                 $this->options[$v] = $v;
 82                 unset( $this->options[$k] );
 83             }
 84         }
 85 
 86         if ( $this->use_ajax ) {
 87             add_action( 'wp_ajax_' . $this->get_ajax_action(), array( $this, 'autocomplete_search' ) );
 88         }
 89     }

101     /**
102      * Get available options, optionally filtering by a fragment (e.g. for Autocomplete)
103      * @param string $fragment optional fragment to filter by
104      * @return array, key => value of available options
105      */
106     public function get_items( $fragment = Null ) {
107         if ( !$fragment ) {
108             return $this->options;
109         }
110         $ret = array();
111         foreach ( $this->options as $k => $v ) {
112             if ( strpos( $v, $fragment ) !== False ) $ret[$k] = $v;
113         }
114         return $ret;
115     }

117     /**
118      * Get an action to register by hashing (non cryptographically for speed)
119      * the options that make this datasource unique.

122     public function get_ajax_action() {
123         if ( !empty( $this->ajax_action ) ) return $this->ajax_action;
124         return 'fm_datasource_' . crc32( 'base' . json_encode( $this->options ) . $this->options_callback );

127     /**
128      * AJAX callback to find posts

131     public function autocomplete_search() {
132         // Check the nonce before we do anything
133         check_ajax_referer( 'fm_search_nonce', 'fm_search_nonce' );
134         $items = $this->get_items( sanitize_text_field( $_POST['fm_autocomplete_search'] ) );
135 
136         // See if any results were returned and return them as an array
137         if ( !empty( $items ) ) {
138             echo json_encode( $items ); exit;
139         } else {
140             echo "0";
141         }
142 
143         die();
144     }
