 79     /**
 80      * Render form element
 81      * @param mixed $value
 82      * @return string HTML
 83      */
 84     public function form_element( $value ) {
 85         // Avoid null array errors later.
 86         $value['_image_flags'] = isset( $value['_image_flags'] ) ? $value['_image_flags'] : array();
 87 
 88         $all = $this->flatten_arrays( $value );
 89         $out = '<div id="fm-draggablepost"><div class="post-repository-wrapper">';
 90         foreach ( $this->repositories as $name => $repo ) {
 91             $out .= sprintf( '<h2>%s</h2><ul class="post-repository sortables">', $repo['label'] );
 92             if ( isset( $repo['callback'] ) ) {
 93                 $ids = call_user_func( $repo['callback'], $name, $repo );
 94                 if ( !empty( $ids ) ) {
 95                     // Return value here should be all numeric post ids, but we'll tolerate non-numerics if they show up.
 96                     foreach ( $ids as $id ) {
 97                         if ( !is_numeric( $id ) || in_array( $id, $all ) ) {
 98                             continue;
 99                         }
100                         $out .= $this->draggable_item_html( $id );
101                     }
102                 }
103             }
104             else {
105                 $query_args = array(
106                     'post_type' => $repo['post_type'],
107                     'post_status' => 'publish',
108                     'posts_per_page' => $repo['length'],
109                     'orderby' => $repo['orderby'],
110                     'order' => $repo['order'],
111                     'tax_query' => array($repo['taxonomy_args']),
112                 );
113                 $q = new WP_Query( $query_args );
114                 while ( $q->have_posts() ) {
115                     $q->the_post();
116                     if ( in_array( get_the_ID(), $all ) ) {
117                         continue;
118                     }
119                     $out .= $this->draggable_item_html( get_the_ID() );
120                 }
121             }
122             $out .= "</ul>";
123         }
124         $out .= "</div>";
125         $out .= '<div class="post-bin-wrapper">';
126         foreach ( $this->bins as $name => $bin ) {
127             $out .= sprintf( '<h2>%s</h2>', $bin );
128             $out .= sprintf( '<ul class="post-bin sortables" id="%s-bin"><em class="empty-message">drop posts here</em>', $name );
129             if ( isset( $value[$name] ) ) {
130                 foreach ( $value[$name] as $id ) {
131                     if ( !$id ) {
132                         continue;
133                     }
134                     $out .= $this->draggable_item_html( $id, in_array( $id, $value['_image_flags'] ) );
135                 }
136             }
137             $out .= "</ul>";
138         }
139 
140         foreach ( $this->bins as $bin => $label ) {
141             $out .= sprintf( '<input type="hidden" value="%s" name="%s" id="%s" />', 
142                         empty( $value[$bin] ) ?  '' : implode( ',', $value[$bin] ),
143                         $this->get_form_name() . '[' . $bin . ']',
144                         $bin
145                     );
146         }
147         $out .= '</div></div>';
148         return $out;
149     }

151     /**
152      * Generate the html for a single draggable item
153      * @param int $post_id
154      * @param boolean $use_image_checked if true, render this item with the "use image" checkbox checked (if enabled)
155      * @return string containing the li element.
156      */
157     protected function draggable_item_html( $post_id, $use_image_checked = false ) {
158         $post = get_post( $post_id );
159         $bylines = array();
160         if ( is_plugin_active( 'co-authors-plus/co-authors-plus.php' ) ) {
161             $authors = get_coauthors( $post_id );
162             foreach ($authors as $author) {
163                 $bylines[] = $author->display_name;
164             }
165             if ( empty( $bylines ) ) {
166                 $authorstr = '(no authors)';
167             }
168             else {
169                 $authorstr = implode(', ', $bylines);
170             }
171         }
172         else {
173             $author = get_userdata( $post->post_author );
174             $authorstr = $author->display_name;
175         }
176 
177         $image_meta = get_post_meta( $post_id, '_thumbnail_id' );
178 
179         $permalink = get_permalink( $post_id );
180 
181         if ( isset( $image_meta[0] ) ) {
182             $image = wp_get_attachment_image( $image_meta[0], array(32,32) );
183         }
184         else{
185             $image = '';
186         }
187 
188         $li = sprintf( '<li class="draggable-post" id="draggable-post-%d" post_id="%d">', $post_id, $post_id );
189         $li_inner = sprintf('<strong><a href="%s" target="_new">%s</a></strong><br />
190                             <small>%s &mdash; %s</small><br />
191                             <small><em>%s %s</em></small>',
192             $permalink,
193             $post->post_title,
194             $post->post_date,
195             $authorstr,
196             $image,
197             $post->post_excerpt
198         );
199 
200         if ( $this->use_image_checkbox && $image ) {
201             $checked = $use_image_checked ? 'checked' : '';
202             $li_inner .= sprintf( '<small><input type="checkbox" value="1" name="%s[_image_flags][%d]" %s /> Use image?</small>', $this->get_form_name(), $post_id, $checked );
203         }
204 
205         $li = $li . apply_filters( 'fieldmanager_draggablepost_li_content', array('li_inner' => $li_inner, 'post_id' => $post_id) ) . '</li>';
206         return $li;
207     }

209     /**
210      * Helper to convert the value passed to form_element into a non-hierarchical list of post_ids so we know which posts are already assigned to a bin and thus should be skipped in rendering.
211      * @param array $value as passed to form_element()
212      * @return array containing post_ids in any subarray, except the one attached to key '_image_flags'.
213      */
214     protected function flatten_arrays( $value ) {
215         if ( empty ( $value ) ) {
216             return array();
217         }
218         $result = array();
219         foreach ( $value as $key => $array ) {
220             if ( $key == '_image_flags' ) {
221                 continue;
222             }
223             $result = array_merge( $result, $value[$key] );
224         }
225         $return = array();
226         foreach ( $result as $id ) {
227             if ( $id ) {
228                 $return[] = $id;
229             }
230         }
231         return $return;
232     }
