 24     /**
 25      * Create page context handler.
 26      * @param string unique form ID
 27      * @param Fieldmanager_Field $fm
 28      */
 29     public function __construct( $uniqid, $fm ) {
 30         $this->fm = $fm;
 31         self::$forms[$uniqid] = $this;
 32         $this->uniqid = $uniqid;
 33 
 34         // since this should be set up in init, check for submit now
 35         if ( !empty( $_POST ) && ! empty( $_POST['fm-page-action'] ) && esc_html( $_POST['fm-page-action'] ) == $uniqid ) {
 36             $this->save_page_form();
 37         }

 40     /**
 41      * Action to save the form
 42      * @return void
 43      */
 44     public function save_page_form() {
 45         if( !wp_verify_nonce( $_POST['fieldmanager-' . $this->fm->name . '-nonce'], 'fieldmanager-save-' . $this->fm->name ) ) {
 46             $this->fm->_unauthorized_access( 'Nonce validation failed' );
 47         }
 48         $this->fm->data_id = $user_id;
 49         $value = isset( $_POST[ $this->fm->name ] ) ? $_POST[ $this->fm->name ] : "";
 50         if ( empty( $this->fm->data_type ) ) $this->fm->data_type = 'page';
 51         if ( empty( $this->fm->data_id ) ) $this->fm->data_id = $this->uniqid;
 52         $current = apply_filters( 'fm_' . $this->uniqid . '_load', array(), $this->fm );
 53         $data = apply_filters( 'fm_' . $this->uniqid . '_presave', $value, $this->fm );
 54         $data = $this->fm->presave_all( $data, $current );
 55         $data = apply_filters( 'fm_presave_data', $data, $this->fm );
 56         do_action( 'fm_' . $this->uniqid . '_save', $data, $current, $this->fm );
 57         $this->did_save = True;
 58     }
 59 
 60     /**
 61      * Output HTML for the form
 62      * @return void
 63      */
 64     public function render_page_form() {
 65         $current = apply_filters( 'fm_' . $this->uniqid . '_load', array(), $this->fm );
 66         echo '<form method="POST" id="' . esc_attr( $this->uniqid ) . '">';
 67         echo '<div class="fm-page-form-wrapper">';
 68         printf( '<input type="hidden" name="fm-page-action" value="%s" />', sanitize_title( $this->uniqid ) );
 69         wp_nonce_field( 'fieldmanager-save-' . $this->fm->name, 'fieldmanager-' . $this->fm->name . '-nonce' );
 70         echo $this->fm->element_markup( $current );
 71         echo '</div>';
 72         printf( '<input type="submit" name="fm-submit" class="button-primary" value="%s" />', esc_attr( $this->fm->submit_button_label ) ?: __( 'Save Options' ) );
 73         echo '</form>';
 74         echo '</div>';
 75         
 76         // Check if any validation is required
 77         $fm_validation = Fieldmanager_Util_Validation( $this->uniqid, 'page' );
 78         $fm_validation->add_field( $this->fm );
 79     }
