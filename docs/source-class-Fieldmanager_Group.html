 99         // Convenient naming of child elements via their keys
100         foreach ( $this->children as $name => $element ) {
101             // if the array key is not an int, and the name attr is set, and they don't match, we got a problem.
102             if ( $element->name && !is_int( $name ) && $element->name != $name ) throw new FM_Developer_Exception( __( 'Group child name conflict: ' ) . $name . ' / ' . $element->name );

114     /**
115      * Render the element, iterating over children and calling their form_element() functions.
116      * @param mixed $value
117      */
118     public function form_element( $value = NULL ) {
119         $out = '';
120         $tab_group = '';
121         $tab_group_submenu = '';
122 
123         // We do not need the wrapper class for extra padding if no label is set for the group
124         if ( isset( $this->label ) && !empty( $this->label ) ) $out .= '<div class="fm-group-inner">';
125 
126         // If the display output for this group is set to tabs, build the tab group for navigation
127         if ( $this->tabbed ) $tab_group = sprintf( '<ul class="fm-tab-bar wp-tab-bar" id="%s-tabs">', $this->get_element_id() );
128 
129         // Produce HTML for each of the children
130         foreach ( $this->children as $element ) {
131 
132             $element->parent = $this;
133 
134             // If the display output for this group is set to tabs, add a tab for this child
135             if ( $this->tabbed ) {
136 
137                 // Set default classes to display the first tab content and hide others
138                 $tab_classes = array( 'fm-tab' );
139                 $tab_classes[] = ( $this->child_count == 0 ) ? "wp-tab-active" : "hide-if-no-js";
140 
141                 // Generate output for the tab. Depends on whether or not there is a tab limit in place.
142                 if ( $this->tab_limit == 0 || $this->child_count < $this->tab_limit ) {
143                     $tab_group .=  sprintf( '<li class="%s"><a href="#%s-tab">%s</a></li>',
144                         implode( " ", $tab_classes ),
145                         $element->get_element_id(),
146                         $element->label
147                      );
148                 } else if ( $this->tab_limit != 0 && $this->child_count >= $this->tab_limit ) {
149                     $submenu_item_classes = array( 'fm-submenu-item' );
150                     $submenu_item_link_class = "";
151 
152                     // Create the More tab when first hitting the tab limit
153                     if ( $this->child_count == $this->tab_limit ) {
154                         // Create the tab
155                         $tab_group_submenu .=  sprintf( '<li class="fm-tab fm-has-submenu %s"><a href="#%s-tab">%s</a>',
156                             $tab_class,
157                             $element->get_element_id(),
158                             __( 'More...' )
159                          );
160 
161                          // Start the submenu
162                          $tab_group_submenu .= sprintf(
163                             '<div class="fm-submenu" id="%s-submenu"><div class="fm-submenu-wrap fm-submenu-wrap"><ul>',
164                             $this->get_element_id()
165                          );
166 
167                          // Make sure the first submenu item is designated
168                          $submenu_item_classes[] = 'fm-first-item';
169                          $submenu_item_link_class = 'class="fm-first-item"';
170                     }
171 
172                     // Add this element to the More menu
173                     $tab_group_submenu .=  sprintf( '<li class="%s"><a href="#%s-tab" %s>%s</a></li>',
174                         implode( ' ', $submenu_item_classes ),
175                         $element->get_element_id(),
176                         $submenu_item_link_class,
177                         $element->label
178                     );
179                 }
180 
181                 // Ensure the child is aware it is tab content
182                 $element->is_tab = TRUE;
183             }
184 
185             // Get markup for the child element
186             $child_value = empty( $value[ $element->name ] ) ? Null : $value[ $element->name ];
187 
188             // propagate editor state down the chain
189             if ( $this->data_type ) $element->data_type = $this->data_type;
190             if ( $this->data_id ) $element->data_id = $this->data_id;
191 
192             $out .= $element->element_markup( $child_value );
193 
194             $this->child_count++;
195 
196         }
197 
198         // We do not need the wrapper class for extra padding if no label is set for the group
199         if ( isset( $this->label ) && !empty( $this->label ) ) $out .= '</div>';
200 
201         // If the display output for this group is set to tabs, build the tab group for navigation
202         if ( $this->tab_limit != 0 && $this->child_count >= $this->tab_limit ) $tab_group_submenu .= '</ul></div></div></li>';
203         if ( $this->tabbed ) $tab_group .= $tab_group_submenu . '</ul>';
204 
205 
206         // Return the complete HTML
207         return $tab_group . $out;
208     }

219     /**
220      * Presave override for groups which dispatches to child presave_all methods.
221      * @input mixed[] values
222      * @return mixed[] values
223      */
224     public function presave( $values, $current_values = array() ) {
225         // @SECURITY@ First, make sure all the values we're given are legal.
226         if( isset( $values ) && !empty( $values ) ) {
227             foreach ( array_keys( $values ) as $key ) {
228                 if ( !isset( $this->children[$key] ) ) {
229                     // If we're here, it means that the input, generally $_POST, contains a value that doesn't belong,
230                     // and thus one which we cannot sanitize and must not save. This might be an attack.
231                     $this->_unauthorized_access( sprintf( 'Found "%1$s" in data but not in children', $key ) );
232                 }
233             }
234         }
235 
236         // Then, dispatch them for sanitization to the children.
237         foreach ( $this->children as $k => $element ) {
238             $element->data_id = $this->data_id;
239             $element->data_type = $this->data_type;
240             if ( empty( $values[$element->name] ) ) {
241                 $values[ $element->name ] = NULL;
242             }
243             $child_value = empty( $values[ $element->name ] ) ? Null : $values[ $element->name ];
244             $current_child_value = empty( $current_values[$element->name ]) ? array() : $current_values[$element->name];
245             $values[ $element->name ] = $element->presave_all( $values[ $element->name ], $current_child_value );
246             if ( !$this->save_empty && $this->limit != 1 ) {
247                 if ( is_array( $values[$element->name] ) && empty( $values[$element->name] ) ) unset( $values[$element->name] );
248                 elseif ( empty( $values[$element->name] ) ) unset( $values[$element->name] );
249             }
250         }
251 
252         if ( is_callable( $this->group_is_empty ) ) {
253             if ( call_user_func( $this->group_is_empty, $values ) ) {
254                 $values = array();
255             }
256         }
257 
258         return $values;
259     }

261     /**
262      * Get an HTML label for this element.
263      * @param array $classes extra CSS classes.
264      * @return string HTML label.
265      */
266     public function get_element_label( $classes = array() ) {
267         $classes[] = 'fm-label';
268         $classes[] = 'fm-label-' . $this->name;
269 
270         $wrapper_classes = array( 'fm-group-label-wrapper' );
271 
272         if ( $this->sortable ) {
273             $wrapper_classes[] = 'fmjs-drag';
274             $wrapper_classes[] = 'fmjs-drag-header';
275         }
276 
277         $extra_attrs = '';
278         if ( $this->label_macro ) {
279             $this->label_format = $this->label_macro[0];
280             $this->label_token = sprintf( '.fm-%s input.fm-element', $this->label_macro[1] );
281         }
282 
283         if ( $this->label_format && $this->label_token ) {
284             $extra_attrs = sprintf(
285                 'data-label-format="%1$s" data-label-token="%2$s"',
286                 htmlentities( $this->label_format ),
287                 htmlentities( $this->label_token )
288             );
289             $classes[] = 'fm-label-with-macro';
290         }
291 
292         return sprintf(
293             '<div class="%1$s"><%2$s class="%3$s"%4$s>%5$s</%2$s>%6$s</div>',
294             implode( ' ', $wrapper_classes ),
295             $this->label_element,
296             implode( ' ', $classes ),
297             $extra_attrs,
298             $this->label,
299             $this->limit == 0 ? $this->get_remove_handle() : ''
300         );
301     }
